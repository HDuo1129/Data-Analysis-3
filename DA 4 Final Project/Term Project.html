def load_and_preprocess(file_path, indicator_code):
    df = pd.read_csv(file_path, skiprows=3)
    df = df[df['Country Code'].isin(sub_saharan_countries)]
    df = df[df['Indicator Code'] == indicator_code]
    year_cols = [col for col in df.columns if col.isdigit() and 2003 <= int(col) <= 2022]
    df = df[["Country Code"] + year_cols]
    df_long = pd.melt(df, id_vars=["Country Code"], var_name="Year", value_name=indicator_code)
    df_long["Year"] = df_long["Year"].astype(int)
    return df_long
